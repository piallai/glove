119template <class Tparam>
120std::string GlvParameterWidget<Tparam>::get_rules_description(const SlvParameter<Tparam>& _parameter) const {
121
122 std::string description;
123
124 std::vector<std::string> descriptions = _parameter.get_rules_description();
125
126 for (typename std::vector<std::string>::const_iterator it = descriptions.begin(); it != descriptions.end(); ++it) {
127 description += *it;
128 if (std::next(it) != descriptions.end() && !it->empty()) {
129 description += "\n";
130 }
131 }
132
133 return description;
