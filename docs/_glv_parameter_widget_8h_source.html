123template <class Tparam>
124std::string GlvParameterWidget<Tparam>::get_rules_description(const SlvParameter<Tparam>& _parameter) const {
125
126 std::string description;
127
128 std::vector<std::string> descriptions = _parameter.get_rules_description();
129
130 for (typename std::vector<std::string>::const_iterator it = descriptions.begin(); it != descriptions.end(); ++it) {
131 description += *it;
132 if (std::next(it) != descriptions.end() && !it->empty()) {
133 description += "\n";
134 }
135 }
136
