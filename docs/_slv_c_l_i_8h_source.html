127template <class Tparametrization>
128SlvStatus SlvCLI::parse(Tparametrization& _parametrization, Arguments& _arguments) {
129
130 SlvStatus status;
131
132 if (!_arguments.is_empty()) {
133
134 std::map<std::string, std::string> stream_values;
135 for (Arguments::Tparameters::const_iterator it = _arguments.get_parameter_arguments().begin(); it != _arguments.get_parameter_arguments().end(); ++it) {
136 stream_values[it->first] = it->second[0];
137 }
138
139 for (std::vector<std::string>::const_iterator it = _arguments.get_solo_arguments().begin(); it != _arguments.get_solo_arguments().end(); ++it) {
140 stream_values[*it] = "1";
141 }
142
143 std::pair< std::map<std::string, int>, std::vector<std::string> > conflicts_missing = _parametrization.set_stream_values(stream_values, false);
144 if (!conflicts_missing.first.empty()) {
145 status += SlvStatus(SlvStatus::statusType::warning, "Multiple parameter correspondences in parametrization.");
146
147 for (std::map<std::string, int>::const_iterator it = conflicts_missing.first.begin(); it != conflicts_missing.first.end(); ++it) {
148
149 status.add_sub_status(SlvStatus(SlvStatus::statusType::warning, it->first + " : " + slv::string::to_string(it->second) + " correspondences"));
150
151 }
152
153 }
154
155 _arguments.filter(conflicts_missing.second);
156
157 }
158
159 return status;
160}

162template <class Tparametrization>
163std::string SlvCLI::get_CLI(const Tparametrization& _parametrization, bool _l_CLI_mode) {
164
165 std::string CLI_line;
166
167 std::pair< std::vector< std::pair<std::string, std::string> >, std::vector< std::pair<std::string, bool> > > serialization = _parametrization.get_string_serialization_bool();
168
169 if (_l_CLI_mode) {
170 serialization = _parametrization.get_string_serialization_bool();
171 } else {
172 serialization.first = _parametrization.get_string_serialization();
173 }
174
175 std::pair<int, char**> cli_arguments = SlvCLI::get_arguments(serialization.first, serialization.second, false);
176
177 for (int i = 1; i < cli_arguments.first; i++) {
178 CLI_line += cli_arguments.second[i];
179 if (i < cli_arguments.first - 1) {
180 CLI_line += " ";
181 }
182 }
183
184 return CLI_line;
185}
