104template <class Tparametrization>
105SlvStatus SlvCLI::parse(Tparametrization& _parametrization, Arguments& _arguments) {
106
107 SlvStatus status;
108
109 if (!_arguments.is_empty()) {
110
111 std::map<std::string, std::string> stream_values;
112 for (Arguments::Tparameters::const_iterator it = _arguments.get_parameter_arguments().begin(); it != _arguments.get_parameter_arguments().end(); ++it) {
113 stream_values[it->first] = it->second[0];
114 }
115
116 for (std::vector<std::string>::const_iterator it = _arguments.get_solo_arguments().begin(); it != _arguments.get_solo_arguments().end(); ++it) {
117 stream_values[*it] = "1";
118 }
119
120 std::pair< std::map<std::string, int>, std::vector<std::string> > conflicts_missing = _parametrization.set_stream_values(stream_values, false);
121 if (!conflicts_missing.first.empty()) {
122 status += SlvStatus(SlvStatus::statusType::warning, "Multiple parameter correspondences in parametrization.");
123
124 for (std::map<std::string, int>::const_iterator it = conflicts_missing.first.begin(); it != conflicts_missing.first.end(); ++it) {
125
126 status.add_sub_status(SlvStatus(SlvStatus::statusType::warning, it->first + " : " + slv::string::to_string(it->second) + " correspondences"));
127
128 }
129
130 }
131
132 _arguments.filter(conflicts_missing.second);
133
134 }
135
136 return status;
137}
