127
128 SlvStatus status;

143SlvStatus SlvFileMgr::read_binary(Tdat& _dat, const std::string& _file_path) {
144
145 SlvStatus status;
146
147 std::ifstream input_file;
148 status = SlvFileMgr::open_file(input_file, _file_path, std::ios::in, std::ios::binary);
149
150 if (status) {
151 bool l_read = slv::rw::readB(_dat, input_file);
152 input_file.close();

154 }

218template <class Tfile>
219SlvStatus SlvFileMgr::open_file(Tfile& _file_stream, const SlvFile& _file, std::ios::openmode _mode1, std::ios::openmode _mode2, std::ios::openmode _mode3) {
220
221 SlvStatus status;
222
223 _file_stream.open(SlvFileMgr::get_path(_file).c_str(), _mode1 | _mode2 | _mode3);
224 if (!_file_stream.is_open()) {
225 const int max_errorlen = 95;// no access to strerrorlen_s
226 char* errmsg = new char[max_errorlen];
227#if defined(OS_WIN)
228 strerror_s(errmsg, max_errorlen, errno);
229#elif defined(OS_LINUX) || defined(OS_UNIX)
230 errmsg = strerror_r(errno, errmsg, max_errorlen);
231#endif
232 status = SlvStatus(SlvStatus::statusType::information, std::string(errmsg) + ", couldn't open file: " + SlvFileMgr::get_path(_file));
233 delete[] errmsg;
234 }
235
236 return status;
237
