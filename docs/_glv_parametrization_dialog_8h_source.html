153template <class Tparametrization>
154bool GlvParametrizationDialog<Tparametrization>::abide_rules() {
155
156 // Update parametrization
157 update_parametrization();
158 // Abide rules
159 bool l_abide = parametrization_base->abide_rules();
160
161 // If rules were not abided (parametrization was modified)
162 if (!l_abide) {
163 // Assign to widget
164 dynamic_cast<GlvParametrizationWidget<Tparametrization>*>(parametrization_widget)->set_parametrization(*dynamic_cast<Tparametrization*>(parametrization_base));
165 }
166
167 return l_abide;
168}
