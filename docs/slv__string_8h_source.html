122template <class T>
123std::string slv::string::extract_str_up_to(const std::string& _string, const T _stop) {
124
125 std::string::size_type pos = _string.find(_stop);
126 if (pos != std::string::npos) {
127 return _string.substr(0, pos);
128 } else {
129 return _string;
130 }
131}

133template <class T>
134T slv::string::read_data_line(std::string& _string_line, const std::string _separator) {
135
136 std::string data_string = extract_str_up_to(_string_line, _separator);
137 T data = string_to_value<T>(data_string);
138 _string_line.erase(0, data_string.size() + _separator.size());
139 return data;
140
141}
