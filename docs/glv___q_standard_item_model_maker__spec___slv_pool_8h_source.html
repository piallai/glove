74 static SlvPool<Tvalue, Tlabel> get_value(const QStandardItemModel* _model, const QModelIndex& _index, const int _Mrows, const int _Ncols) {
75
76 SlvPool<Tvalue, Tlabel> pool;
77
78 if (_index == QModelIndex() || _index.model() == _model) {
79
80 int N = (_Mrows == -1) ? _model->rowCount(_index) : std::min(_model->rowCount(_index), _Mrows);
81
82 if (_model->columnCount(_index) >= 2) {
83
84 QStandardItem* item;
85 Tvalue* value;
86 for (int i = 0; i < N; i++) {
87 item = _model->itemFromIndex(_model->index(i, 0, _index));
88 value = pool.new_element(glv::tdata::fromQStandardItem<Tlabel>(item));
89 item = _model->itemFromIndex(_model->index(i, 1, _index));
90 *value = glv::tdata::fromQStandardItem<Tvalue>(item);
91 }
92
93 } else {
94 slv::flag::ISSUE(slv::flag::Critical, "SlvPool requires two columns in the model");
95 }
96
97 } else {
98 slv::flag::ISSUE(slv::flag::Critical, "index's model is different from the one it is supposed to belong to");
99 }
100
101 return pool;
102 }
