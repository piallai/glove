126template <class T>
127bool slv::vector::remove(const T& _element, std::vector<T>& _elements) {
128
129#if VECTOR_EXPLICIT_ALGORITHM==1
130 bool l_found = false;
131 unsigned int el = 0;
132 while (!l_found && el < _elements.size()) {
133 l_found = (_element == _elements[el]);
134 el++;
135 }
136 if (l_found) {
137 el--;
138 _elements.erase(_elements.begin() + el);
139 }
140 return l_found;
141#else
142
143 typename std::vector<T>::iterator it = std::find(_elements.begin(), _elements.end(), _element);
144 if (it != _elements.end()) {
145 _elements.erase(it);
146 return true;
147 } else {
148 return false;
149 }
150#endif
151}

386template <class T>
387bool slv::vector::equalUnordered(const std::vector<T>& _vector1, const std::vector<T>& _vector2) {
388
389 if (_vector1.size() != _vector2.size()) {
390 return false;
391 } else {
392 unsigned int i = 0;
393 unsigned int j;
394 std::vector<bool> vector2_already_match(_vector2.size(), false);
395 bool l_equal = true;
396 while (l_equal && i < _vector1.size()) {
397 j = 0;
398 while (j < _vector2.size() && (_vector1[i] != _vector2[j] || vector2_already_match[j])) {
399 j++;
400 }//stops : either if j out of range, either if matching is found (provided not already match)
401 if (j == _vector2.size()) {
402 l_equal = false;//no matching found
403 } else {
404 vector2_already_match[j] = true;
405 i++;
406 }
407 }
408 return l_equal;
409 }

425template <class T>
426void slv::vector::sort_ascending(std::vector<T>& _elements, unsigned int _range) {
427
428#if VECTOR_EXPLICIT_ALGORITHM==1
429 if (_range == 0) {
430 _range = _elements.size();
431 }
432
433 unsigned int el, el2;
434 for (el = 1; el < _range; el++) {
435 el2 = el;
436 while (el2 > 0 && _elements[el2] < _elements[el2 - 1]) {
437 std::swap(_elements[el2 - 1], _elements[el2]);
438 el2--;
439 }
440 }
441#else
442 typename std::vector<T>::iterator it;
443 if (_range != 0) {
444 it = _elements.begin() + _range;
445 } else {
446 it = _elements.end();
447 }
448 std::sort(_elements.begin(), it);
449#endif
450}

534
535#if VECTOR_EXPLICIT_ALGORITHM==1

542 }

546 }

550

552 typename std::vector<T>::const_iterator it2 = _elements2.begin();
553
554 while (l_equal && it1 != _elements1.end()) {
555 l_equal = (*it1 == *it2);
