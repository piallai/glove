125template <class T>
126bool slv::vector::remove(const T& _element, std::vector<T>& _elements) {
127
128#if VECTOR_EXPLICIT_ALGORITHM==1
129 bool l_found = false;
130 unsigned int el = 0;
131 while (!l_found && el < _elements.size()) {
132 l_found = (_element == _elements[el]);
133 el++;
134 }
135 if (l_found) {
136 el--;
137 _elements.erase(_elements.begin() + el);
138 }
139 return l_found;
140#else
141
142 typename std::vector<T>::iterator it = std::find(_elements.begin(), _elements.end(), _element);
143 if (it != _elements.end()) {
144 _elements.erase(it);
145 return true;
146 } else {
147 return false;
148 }
149#endif
150}

385template <class T>
386bool slv::vector::equalUnordered(const std::vector<T>& _vector1, const std::vector<T>& _vector2) {
387
388 if (_vector1.size() != _vector2.size()) {
389 return false;
390 } else {
391 unsigned int i = 0;
392 unsigned int j;
393 std::vector<bool> vector2_already_match(_vector2.size(), false);
394 bool l_equal = true;
395 while (l_equal && i < _vector1.size()) {
396 j = 0;
397 while (j < _vector2.size() && (_vector1[i] != _vector2[j] || vector2_already_match[j])) {
398 j++;
399 }//stops : either if j out of range, either if matching is found (provided not already match)
400 if (j == _vector2.size()) {
401 l_equal = false;//no matching found
402 } else {
403 vector2_already_match[j] = true;
404 i++;
405 }
406 }
407 return l_equal;
408 }

424template <class T>
425void slv::vector::sort_ascending(std::vector<T>& _elements, unsigned int _range) {
426
427#if VECTOR_EXPLICIT_ALGORITHM==1
428 if (_range == 0) {
429 _range = _elements.size();
430 }
431
432 unsigned int el, el2;
433 for (el = 1; el < _range; el++) {
434 el2 = el;
435 while (el2 > 0 && _elements[el2] < _elements[el2 - 1]) {
436 std::swap(_elements[el2 - 1], _elements[el2]);
437 el2--;
438 }
439 }
440#else
441 typename std::vector<T>::iterator it;
442 if (_range != 0) {
443 it = _elements.begin() + _range;
444 } else {
445 it = _elements.end();
446 }
447 std::sort(_elements.begin(), it);
448#endif
449}

531template <class T>
532bool slv::vector::is_equal(const std::vector<T>& _elements1, const std::vector<T>& _elements2) {
533
534#if VECTOR_EXPLICIT_ALGORITHM==1
535 if (_elements1.size() == _elements2.size()) {
536 unsigned int i = 0;
537 bool l_ok = true;
538 while (l_ok && i < _elements1.size()) {
539 l_ok = (_elements1[i] == _elements2[i]);
540 i++;
541 }
542 return l_ok;
543 } else {
544 return false;
545 }
546#else
547 bool l_equal = true;
548 if (_elements1.size() == _elements2.size()) {
549
550 typename std::vector<T>::const_iterator it1 = _elements1.begin();
551 typename std::vector<T>::const_iterator it2 = _elements2.begin();
552
553 while (l_equal && it1 != _elements1.end()) {
554 l_equal = (*it1 == *it2);
555 ++it1;
556 ++it2;
557 }
558
559 } else {
560 l_equal = false;
561 }
562 return l_equal;
