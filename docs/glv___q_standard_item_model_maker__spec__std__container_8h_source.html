164template <class Tcontainer>
165void glv::tdata::QStandardItemModelMaker<Tcontainer, Tenable>::make(const Tcontainer& _container, QStandardItemModel* _model, const QModelIndex _index, const Qt::ItemFlag _flag_additional) {
166
167 if (_model) {
168
169 if (_index == QModelIndex() || _index.model() == _model) {
170
171 unsigned int Mrows = (unsigned int)_container.size();
172 glv::resize_rows(_model, Mrows, _index);
173 unsigned int Ncols = (unsigned int)ContainerHandler<Tcontainer>::get_size(_container);
174 glv::resize_cols(_model, Ncols, _index);
175
176 QModelIndex index;
177 typename Tcontainer::const_iterator it = _container.begin();
178 for (int i = 0; i < (int)Mrows; i++, ++it) {
179 ContainerHandler<Tcontainer>::set_model(*it, i, _model, _index, _flag_additional);
180 }
181
182 if (_index != QModelIndex()) {
183 QString root_text = get_root_text(Mrows, Ncols);
184 if (_model->itemFromIndex(_index)->text() != root_text) {
185 _model->itemFromIndex(_index)->setText(root_text);
186 }
187 }
188
189 } else {
190 slv::flag::ISSUE(slv::flag::Critical, "index's model is different from the one it is supposed to belong to");
191 }
192
193 } else {
194 slv::flag::ISSUE(slv::flag::Critical, "_model is NULL");
195 }
196
197}

199template <class Tcontainer>
200Tcontainer glv::tdata::QStandardItemModelMaker<Tcontainer, Tenable>::get_value(const QStandardItemModel* _model, const QModelIndex& _index, const int _Mrows, const int _Ncols) {
201
202 Tcontainer container;
203
204 if (_index == QModelIndex() || _index.model() == _model) {
205
206 int Mrows = (_Mrows == -1) ? _model->rowCount(_index) : std::min(_model->rowCount(_index), _Mrows);
207 int Ncols = (_Ncols == -1) ? _model->columnCount(_index) : std::min(_model->columnCount(_index), _Ncols);
208
209 ContainerHandlerResize<Tcontainer>::resize(container, Mrows);
210 for (typename Tcontainer::iterator it = container.begin(); it != container.end(); ++it) {
211 ContainerHandler<Tcontainer>::resize(*it, Ncols);
212 }
213
214 typename Tcontainer::iterator it = container.begin();
215 for (int i = 0; i < Mrows; i++, ++it) {
216 ContainerHandler<Tcontainer>::get_value(*it, i, _model, _index);
217 }
218
219 } else {
220 slv::flag::ISSUE(slv::flag::Critical, "index's model is different from the one it is supposed to belong to");
221 }
222
223 return container;
224
225}

227template <class Tcontainer>
228std::vector<int> glv::tdata::QStandardItemModelMaker<Tcontainer, Tenable>::get_coordinates(QModelIndex& _index) {
229
230 std::vector<int> coordinates = glv::tdata::QStandardItemMaker<Tdata_sub>::get_coordinates(_index);
231 ContainerHandler<Tcontainer>::add_coordinate(coordinates, _index);
232 coordinates.insert(coordinates.begin(), _index.row());
233 _index = _index.parent();
234
235 return coordinates;
