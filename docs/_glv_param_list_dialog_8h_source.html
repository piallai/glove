116template <class Tlist, class Toption>
117void GlvParamListDialog<Tlist, Toption>::accept() {
118
119 // Do not call new_parametrization_dialog if a parametrization widget already exists.
120 // Avoids getting list items value that are by default not the ones that may be configured
121 // Call it if combo item name differs from the current parametrization (ex: combo item has changed and direct 'ok')
122 if (!parametrization_dialog || parametrization_dialog->get_parametrization_base()->get_name() != combo_list->currentText().toStdString()) {
123 bool l_show = false;
124 new_parametrization_dialog(combo_list->currentText(), l_show);
125 }
126
127 GlvListDialog::accept();
128
129}
