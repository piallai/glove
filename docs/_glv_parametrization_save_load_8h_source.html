76template <class Tparametrization>
77void GlvParametrizationSaveLoad<Tparametrization>::save(const std::string& _file_name) {
78
79 bool l_write_default_binary = true;
80
81#if OPTION_USE_THIRDPARTY_JSON==1
82 if (SlvFileMgr::get_extension(_file_name) == ".json") {
83 if (slv::rw::json::ReadWrite<Tparametrization>::l_valid) {
84
85 l_write_default_binary = false;
86
87 std::ofstream file_stream;
88 SlvStatus status = SlvFileMgr::open_file(file_stream, _file_name);
89 if (status) {
90 nlohmann::json json_value;
91 slv::rw::json::ReadWrite<Tparametrization>::writeJson(parametrization_widget->get_value(), json_value);
92 file_stream << json_value.dump(4);
93 file_stream.close();
94 }
95
96 } else {
97 SlvStatus status_tmp(SlvStatus::statusType::warning, "Json parser is not managed for this type.\nUsing default binary parser.");
98 glv::flag::showQMessageBox(status_tmp, false, this);
99 }
100 }
101#endif
102
103 if (l_write_default_binary) {
104 SlvFileMgr::write_binary(parametrization_widget->get_value().param_cast(), _file_name);
105 }
106
107}

109template <class Tparametrization>
110SlvStatus GlvParametrizationSaveLoad<Tparametrization>::load(const std::string& _file_name) {
111
112 SlvStatus status;
113 Tparametrization value;
114
115 bool l_read_default_binary = true;
116
117#if OPTION_USE_THIRDPARTY_JSON==1
118 if (SlvFileMgr::get_extension(_file_name) == ".json") {
119 if (slv::rw::json::ReadWrite<Tparametrization>::l_valid) {
120
121 l_read_default_binary = false;
122
123 std::ifstream file_stream;
124 status = SlvFileMgr::open_file(file_stream, _file_name);
125 if (status) {
126
127 nlohmann::json json_value;
128 file_stream >> json_value;
129 if (!json_value.empty()) {
130 status = slv::rw::json::ReadWrite<Tparametrization>::readJson(value, json_value);
131 bool l_set_parameters = interactive_load_parameters(_file_name, status);
132 if (l_set_parameters) {
133 value.param_init();
134 parametrization_widget->set_value(value);
135 }
136
137 }
138
139 }
140
141 } else {
142 SlvStatus status_tmp(SlvStatus::statusType::warning, "Json parser is not managed for this type.\nUsing default binary parser.");
143 glv::flag::showQMessageBox(status_tmp, false, this);
144 }
145 }
146#endif
147
148 if (l_read_default_binary) {
149
150 status = SlvFileMgr::read_binary(value.param_cast(), _file_name);
151 if (status) {
152 value.param_init();
153 parametrization_widget->set_value(value);
154 }
155
156 }
157
158 return status;
159}
