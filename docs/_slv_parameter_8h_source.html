231template <class Tparam>
232SlvStatus SlvParameter<Tparam>::check_rules() const {
233 SlvStatus status;
234 for (unsigned int r = 0; r < get_rules().size(); r++) {
235 status += get_rules()[r].is_abided(this);
236 }
237 for (unsigned int r = 0; r < get_dynamic_rules().size(); r++) {
238 status += get_dynamic_rules()[r].is_abided(this);
239 }
240 return status;
241}

243template <class Tparam>
244bool SlvParameter<Tparam>::abide_rules() {
245
246 bool l_abide_global = true;
247
248 // Maximum attemps
249 int Nmax_loops = 20;
250
251 // If one rules need abiding, then all rules are checked again.
252 bool l_abide = false;// for initialization
253 int i = 0;
254 while (!l_abide && i < Nmax_loops) {
255 l_abide = true;
256
257 for (unsigned int r = 0; r < get_rules().size(); r++) {
258 l_abide &= get_rules()[r].abide(this);
259 }
260
261 i++;
262 l_abide_global &= l_abide;//if l_abide is false at some point, l_abide_global will be too
263 }
264
265 return l_abide_global;
266}
