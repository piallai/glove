116int GlvCLI::main(int _argc, char* _argv[]) {
117
118 if (SlvCLI::has_glove(_argc, _argv)) {
119
120 QApplication app(_argc, _argv);
121
122 std::string autosave_file_name = SlvFileMgr::replace_forbidden_file_characters(Tparametrization::name(), '_', true, true);
123#if OPTION_USE_THIRDPARTY_JSON==1
124 autosave_file_name += ".json";
125#endif
126
127 GlvParametrizationDialog<Tparametrization> dialog;
128 GlvParametrizationSaveLoad<Tparametrization>* save_load_widget = new GlvParametrizationSaveLoad<Tparametrization>(dialog.get_parametrization_widget());
129
130 SlvCLI::Arguments arguments(_argc, _argv);
131
132 if (!arguments.get_glove_argument().empty()) {
133
134 save_load_widget->load(arguments.get_glove_argument());
135
136 }
137
138
139 Tparametrization parametrization = dialog.get_parametrization_widget()->get_parametrization();
140 SlvStatus status;
141
142 if (!arguments.is_empty()) {
143
144 status = SlvCLI::parse(parametrization, arguments);
145
146 glv::flag::showQMessageBox(QObject::tr("Arguments conflict"), status, true);
147
148 dialog.set_parametrization(parametrization);
149
150 } else if (SlvFile(autosave_file_name).exists() && arguments.get_glove_argument().empty()) {
151
152 save_load_widget->load(autosave_file_name);
153
154 }
155
156 int result = dialog.exec();
157 if (result == QDialog::Accepted) {
158
159 save_load_widget->save(autosave_file_name);
160
161 SlvDirectory directory(ParamOutput<Tparametrization>::get_path(dialog.get_parametrization()));
162 if (directory.exists()) {
163 save_load_widget->save(SlvFile(directory, autosave_file_name).get_path());
164 }
165
166 std::vector< std::pair<std::string, std::string> > parameter_arguments = dialog.get_parametrization().get_string_serialization_bool().first;
167 for (SlvCLI::Arguments::Tparameters::const_iterator it = arguments.get_parameter_arguments().begin(); it != arguments.get_parameter_arguments().end(); ++it) {
168 parameter_arguments.push_back({ it->first, it->second[0] });
169 }
170
171 std::vector<std::string> solo_arguments = dialog.get_parametrization().get_string_serialization_bool().second;
172 slv::vector::add(solo_arguments, arguments.get_solo_arguments());
173
174
175 std::pair<int, char**> cli_arguments = SlvCLI::get_arguments(parameter_arguments, solo_arguments);
176 cli_arguments.second[0] = _argv[0];
177
178 return glv_cli_main(cli_arguments.first, cli_arguments.second);
179
180 } else {
181 return 0;
182 }
183
184 } else {
185 return glv_cli_main(_argc, _argv);
186 }
