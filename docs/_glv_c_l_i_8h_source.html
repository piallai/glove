83* GLOVE_APP macro
84* ------------------------------
85Transform a program into a GUI application. To be used at the very beginning of the main.
86- GLOVE_APP : by setting -glove as CLI argument
87- GLOVE_APP_AUTO : no need to set -glove => turn program into GUI as default
88- GLOVE_APP_PARAM(Tparametrization) : same as GLOVE_APP, adds CLI arguments by-pass through a parametrization widget defined by Tparametrization.

273int GlvApp::main_recurrent(int _argc, char* _argv[], bool _l_threaded, Interface& _interface, bool _l_recurrent, Trecurrent& _recurrent_var) {
274
275 std::string autosave_file_name = SlvFileMgr::replace_forbidden_file_characters(Tparametrization::name(), '_', true, true);
276#if OPTION_USE_THIRDPARTY_JSON==1
277 autosave_file_name += ".json";
278#endif
279
280 if (progress_mgr()) {
281 progress_mgr()->hide();
282 }
283
284 GlvParametrizationDialog<Tparametrization> dialog;
285 GlvParametrizationSaveLoad<Tparametrization>* save_load_widget = new GlvParametrizationSaveLoad<Tparametrization>(dialog.get_parametrization_widget());
286
287 SlvCLI::Arguments arguments(_argc, _argv);
288
289 if (!arguments.get_glove_argument().empty()) {
290
291 save_load_widget->load(arguments.get_glove_argument());
292
293 }
294
295 Tparametrization parametrization = dialog.get_parametrization_widget()->get_parametrization();
296 SlvStatus status;
297
298 if (!arguments.is_empty()) {
299
300 status = SlvCLI::parse(parametrization, arguments);
301
302 glv::flag::showQMessageBox(QObject::tr("Arguments conflict"), status, true);
303
304 dialog.set_parametrization(parametrization);
305
306 } else if (SlvFile(autosave_file_name).exists() && arguments.get_glove_argument().empty()) {
307
308 save_load_widget->load(autosave_file_name);
309
310 }
311
312 int result;
313 if (Tparametrization::Nparameters() > 0) {
314 result = dialog.exec();
315 } else {
316 result = QDialog::Accepted;
317 }
318
319 if (result == QDialog::Accepted) {
320
321 save_load_widget->save(autosave_file_name);
322
323 SlvDirectory directory(ParamOutput<Tparametrization>::get_path(dialog.get_parametrization()));
324 if (directory.exists()) {
325 save_load_widget->save(SlvFile(directory, autosave_file_name).get_path());
326 }
327
328 std::vector< std::pair<std::string, std::string> > parameter_arguments = dialog.get_parametrization().get_string_serialization_bool().first;
329 for (SlvCLI::Arguments::Tparameters::const_iterator it = arguments.get_parameter_arguments().begin(); it != arguments.get_parameter_arguments().end(); ++it) {
330 parameter_arguments.push_back({ it->first, it->second[0] });
331 }
332
333 std::vector<std::string> solo_arguments = dialog.get_parametrization().get_string_serialization_bool().second;
334 slv::vector::add(solo_arguments, arguments.get_solo_arguments());
335
336
337 std::pair<int, char**> cli_arguments = SlvCLI::get_arguments(parameter_arguments, solo_arguments);
338 cli_arguments.second[0] = _argv[0];
339
340 if (_l_threaded) {
341
342 if (_l_recurrent) {
343 dynamic_cast<RecurrentWrapperT<Tparametrization, Trecurrent>*>(_interface.interactive_wrapper)->set(cli_arguments.first, cli_arguments.second, &_recurrent_var);
344 }
345
346 _interface.future = QtConcurrent::run(&glv_cli_main<Tparametrization, Trecurrent>, cli_arguments.first, cli_arguments.second, true, dialog.get_parametrization(), _l_recurrent, std::ref(_recurrent_var));
347 _interface.future_watcher.setFuture(_interface.future);
348
349 if (progress_mgr()) {
350 progress_mgr()->show();
351 }
352
353 return 0;
354
355 } else {
356 status_mgr() = NULL;
357 return glv_cli_main(cli_arguments.first, cli_arguments.second, true, dialog.get_parametrization(), _l_recurrent, _recurrent_var);
358 }
359
360 } else {
361 QCoreApplication::quit();
362 return 1;
363 }

368template <class Tparametrization, class Trecurrent>
369int GlvApp::main(int _argc, char* _argv[], bool _l_auto_glove, bool _l_threaded, bool _l_recurrent, Trecurrent& _recurrent_var) {
370
371 if (SlvCLI::has_glove(_argc, _argv) || _l_auto_glove) {
372
373 QApplication q_app(_argc, _argv);
374 if (_l_recurrent) {
375 q_app.setQuitOnLastWindowClosed(false);
376 }
377
378 if (_l_threaded) {
379
380 status_mgr() = new GlvStatusMgr;
381 status_mgr()->add(&GlvApp::status());
382
383 if (!progressions().empty()) {
384 progress_mgr() = new GlvProgressMgr;
385 for (int i = 0; i < progressions().psize(); i++) {
386 bool l_hide_when_over = !progressions()[i]->is_recurrent();
387 progress_mgr()->add_progression(progressions()[i], l_hide_when_over);
388 }
389 }
390
391 if (!_l_recurrent) {
392 QObject::connect(&interface().future_watcher, &QFutureWatcher<int>::finished, &q_app, &QCoreApplication::quit, Qt::QueuedConnection);
393 } else {
394 RecurrentWrapperT<Tparametrization, Trecurrent>* interactive_wrapper = new RecurrentWrapperT<Tparametrization, Trecurrent>;
395 interactive_wrapper->set_interface(&interface());
396 interface().interactive_wrapper = interactive_wrapper;
397 QObject::connect(&interface().future_watcher, SIGNAL(finished()), interface().interactive_wrapper, SLOT(relaunch()));
398 }
399
400 }
401
402 int exit = main_recurrent<Tparametrization>(_argc, _argv, _l_threaded, interface(), _l_recurrent, _recurrent_var);
403
404 if (!exit && _l_threaded) {
405
406 return q_app.exec();
407
408 } else {
409
410 return exit;
411 }
412
413 } else {
414 return glv_cli_main(_argc, _argv, false, Tparametrization(), false, _recurrent_var);
415 }
