60template <class Tlist, template <class Tlist_item, class Toption, class _Tenable> class Tfiltering, class Toption>
61SlvStatus GlvParamListDialog_Filtering::filter_list(QComboBox* _combo_list, int* _Nitems) {
62
63 SlvStatus status;
64 std::string item_name;
65 bool l_enabled;
66 if (_Nitems) *_Nitems = 0;
67 for (int i = 0; i < _combo_list->count(); i++) {
68
69 item_name = _combo_list->itemText(i).toStdString();
70 l_enabled = Spec<Tlist>::template filter<Tfiltering, Toption>(item_name, _combo_list, i);
71
72 if (l_enabled && _Nitems) (*_Nitems)++;
73 }
74
75 update_current_item(_combo_list);
76
77 return status;
78}

90
91 if (!Tfiltering<T1, T2>::is_compatible) {
