90template <class Tdata, template <class _Tdata> class Twidget_final, bool IsPersistent>
91Tdata GlvTableView<Tdata, Twidget_final, IsPersistent>::get_value() const {
92
93 Tdata value;
94 QStandardItemModel* model_ = dynamic_cast<QStandardItemModel*>(model());
95 int Mrows = model_->rowCount();
96 int Ncols = model_->columnCount();
97 if (l_dummy_row) {
98 Mrows--;
99 }
100 if (l_dummy_column) {
101 Ncols--;
102 }
103 value = glv::tdata::fromQStandardItemModel<Tdata>(model_, QModelIndex(), Mrows, Ncols);
104
105 return value;
106}

108template <class Tdata, template <class _Tdata> class Twidget_final, bool IsPersistent>
109void GlvTableView<Tdata, Twidget_final, IsPersistent>::set_value(const Tdata& _data) {
110
111 if (!model()) {
112 setModel(glv::tdata::toQStandardItemModel(_data));
113 } else {
114
115 this->setCurrentIndex(QModelIndex());// Close delegate editor if opened. Prevents overwriting of model when closing the editor after value is set.
116
117 QStandardItemModel* _model = dynamic_cast<QStandardItemModel*>(model());
118 reset_dummies();//must be set before, because model modification can trigger 'parent' of this if this is delegate
119 _model->blockSignals(true);//block signals so cell_changed() is not emitted each time an item of model is changed
120 glv::tdata::toQStandardItemModel(_data, _model);
121 _model->blockSignals(false);
122 resize_fit(true);
123 emit _model->layoutChanged();//to refresh view automatically
124 }
125
126 if (GlvItemDelegate<typename glv::tdata::QStandardItemModelMaker<Tdata>::Tdata_sub, GlvTableView, Twidget_final, IsPersistent>::is_persistent()) {
127 int Mrows = model()->rowCount();
128 int Ncols = model()->columnCount();
129 QModelIndex index;
130 //setUpdatesEnabled(false);
131 for (int i = 0; i < Mrows; ++i) {
132 for (int j = 0; j < Ncols; ++j) {
133 index = model()->index(i, j);
134 bool l_type_mismatch = static_cast<QStandardItemModel*>(model())->itemFromIndex(index)->flags() & glv::tdata::type_mismatch_flag;
135 if (!QTableView::isPersistentEditorOpen(index) && !l_type_mismatch && IsPersistent) {
136 QTableView::openPersistentEditor(index);
137 }
138
139 // If type mismatch : disable editing. Using a type dedicated widget is not possible
140 if (l_type_mismatch && !std::is_same<Twidget_final<void>, GlvLineEdit<void>>::value) {//Twidget_final comparison is actually redundant with is_persistent() for now
141 if (!glv::tdata::l_allow_type_mismatch_edit) {
142 Qt::ItemFlags flags = static_cast<QStandardItemModel*>(model())->itemFromIndex(index)->flags();
143 static_cast<QStandardItemModel*>(model())->itemFromIndex(index)->setFlags(flags ^ Qt::ItemIsEditable);
144 }
145 }
146 }
147 }
148 //setUpdatesEnabled(true);
149 }
150
151
152 if (GlvItemDelegate<typename glv::tdata::QStandardItemModelMaker<Tdata>::Tdata_sub, GlvTableView, Twidget_final, IsPersistent>::is_persistent() && IsPersistent) {
153 //horizontalHeader()->setSectionResizeMode(QHeaderView::ResizeToContents);
154 int rowCount = model()->rowCount();
155 int columnCount = model()->columnCount();
156 if (rowCount > 0) {
157 for (int j = 0; j < columnCount; j++) {
158 QModelIndex index = model()->index(0, j);
159 bool l_type_mismatch = static_cast<QStandardItemModel*>(model())->itemFromIndex(index)->flags() & glv::tdata::type_mismatch_flag;
160 if (!l_type_mismatch) {
161 horizontalHeader()->setSectionResizeMode(j, QHeaderView::ResizeToContents);
162 } else {
163 //horizontalHeader()->setSectionResizeMode(j, QHeaderView::Interactive);
164 }
165 }
166 } else {
167 //horizontalHeader()->setSectionResizeMode(QHeaderView::ResizeToContents);
168 }
169
170 verticalHeader()->setSectionResizeMode(QHeaderView::ResizeToContents);
171 }
172
173}
