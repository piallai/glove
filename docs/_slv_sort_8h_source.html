127template <class Tdata, class Trange>
128unsigned int SlvSort<Tdata, Trange>::get_index(const Tdata& _data) const {
129
130 unsigned int i = 0;
131 unsigned int i_lower = 0;
132 unsigned int i_upper = (unsigned int)datas.size();
133
134 while (i_lower + 1 < i_upper) {
135
136 i = i_lower + (i_upper - i_lower) / 2;
137
138 if (test_data(datas[i], _data)) {
139 i_lower = i;
140 } else {
141 i_upper = i;
142 }
143
144 }
145
146 if (i_lower != i_upper) {
147 if (test_data(datas[i_lower], _data)) {
148 i = i_upper;
149 } else {
150 i = i_lower;
151 }
152 }
153
154 return i;
155
156}

158template <class Tdata, class Trange>
159bool SlvSort<Tdata, Trange>::add(const Tdata& _data) {
160
161 bool l_return = true;
162
163 //get index where _data should belong
164 unsigned int i = get_index(_data);
165
166 if (i == 0) {
167
168 datas.insert(datas.begin(), _data);
169 if (datas.size() > get_Ndatas_max()) {
170 datas.pop_back();
171 } else if (get_range() != Trange(0)) {// range reference may have changed
172 while (!check_range(datas.back())) {
173 datas.pop_back();
174 }
175 }
176
177 } else {
178
179 if (i == datas.size()) {
180
181 if (i < get_Ndatas_max() && (get_range() == Trange(0) || check_range(_data))) {
182 datas.push_back(_data);
183 } else {
184 l_return = false;
185 }
186
187 } else {
188
189 datas.insert(datas.begin() + i, _data);
190 if (datas.size() > get_Ndatas_max()) {
191 datas.pop_back();
192 }
193
194 }
195
196 }
197
198 return l_return;
199}
